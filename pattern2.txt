#include <iostream>
using namespace std;
int main()
{
    int i,j,k,x,n;
    cin>>n;

    for(i=1;i<=n;i++)
    {
        k=1;
        x=i;
        for(j=1;j<=(2*n -1);j++)
        {
            if(j>=i && j<=(2*n -i) && k)
            {
                cout<<x;
                k=0;
                x++;

            }
            else{
                cout<<" ";
                k=1;
            }
        }
        cout<<endl;
    }

     for(i=n+1;i<=(2*n -1);i++)
    {
        k=1;
        x=(2*n-i);
        for(j=1;j<=(2*n-1);j++)
        {
            if(j>=(2*n-i) && j<=i && k)
            {
                cout<<x;
                k=0;
                x++;

            }
            else{
                cout<<" ";
                k=1;
            }
        }
        cout<<endl;
    }

}
